function setupSheets() {

  if (!sheetD) {
    sheetD = ss.insertSheet(SHEET_DEBTORS);

    sheetD.appendRow(['ID', 'Name', 'Moo', 'Date', 'Bottles', 'Amount', 'Status', 'Timestamp']);
    sheetD.getRange("A1:H1").setFontWeight("bold").setBackground("#2563eb").setFontColor("white");
    sheetD.setFrozenRows(1);
  } else {
    // เช็คว่าถ้ายังไม่มีหัวข้อ Moo ให้แทรก

      sheetD.insertColumnAfter(2);
      sheetD.getRange("C1").setValue("Moo");

  }

  let sheetU = ss.getSheetByName(SHEET_USERS);

function getDebtorsData() {
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(SHEET_DEBTORS);
  const data = sheet.getDataRange().getValues();
  if (data.length <= 1) return JSON.stringify([]);
  
  // ปรับให้อ่านค่าจากตารางใหม่ที่มี Moo เข้ามาด้วย
  const result = data.slice(1).map(row => {
    return { id: row[0], name: row[1], moo: row[2], date: row[3], bottles: row[4], amount: row[5], status: row[6], created: row[7] };
  });
  return JSON.stringify(result);
}

function saveDebtorData(jsonData) {
  const data = JSON.parse(jsonData);
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(SHEET_DEBTORS);
  const allData = sheet.getDataRange().getValues();
  
  let isEdit = false;
  for (let i = 1; i < allData.length; i++) {
    if (allData[i][0] === data.id) {
      sheet.getRange(i + 1, 2).setValue(data.name);
      sheet.getRange(i + 1, 3).setValue(data.moo); // เพิ่ม Moo
      sheet.getRange(i + 1, 4).setValue(data.date);
      sheet.getRange(i + 1, 5).setValue(data.bottles);
      sheet.getRange(i + 1, 6).setValue(data.amount);
      isEdit = true; break;
    }
  }
  if (!isEdit) {
    sheet.appendRow([data.id, data.name, data.moo, data.date, data.bottles, data.amount, data.status, new Date().toISOString()]);
  }
  return getDebtorsData();
}

function updateDebtorStatus(id, newStatus) {
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(SHEET_DEBTORS);

      sheet.getRange(i + 1, 7).setValue(newStatus); // เปลี่ยนช่องที่ 7
      break;
    }
  }

}

function deleteDebtorData(id) {
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(SHEET_DEBTORS);

// 1. ฟังก์ชันเช็คและสร้างชีต Deliveries อัตโนมัติ
function getDeliveriesSheet() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Deliveries");
  
  // ถ้ายังไม่มีชีตนี้ ให้สร้างใหม่และใส่หัวคอลัมน์อัตโนมัติ
  if (!sheet) {
    sheet = ss.insertSheet("Deliveries");
    sheet.appendRow(["id", "type", "month", "date", "agency", "plate", "senders", "load", "return", "price", "actual", "total", "timestamp"]);
    
    // ทำตัวหนาให้หัวตาราง
    sheet.getRange("A1:M1").setFontWeight("bold").setBackground("#f3f4f6");
  }
  return sheet;
}

// 2. ฟังก์ชันดึงข้อมูลทั้งหมดมาแสดงในระบบ
function getDeliveriesData() {
  var sheet = getDeliveriesSheet();
  var data = sheet.getDataRange().getValues();
  var result = [];
  
  if (data.length > 1) {
    var headers = data[0];
    for (var i = 1; i < data.length; i++) {
      var row = data[i];
      var obj = {};
      for (var j = 0; j < headers.length; j++) {
        var val = row[j];
        // แปลงข้อความรายชื่อคนส่ง ให้กลับเป็น Array เพื่อให้ระบบหน้าบ้านอ่านได้
        if (headers[j] === 'senders' && typeof val === 'string') {
          obj[headers[j]] = val.split(','); 
        } else {
          obj[headers[j]] = val;
        }
      }
      result.push(obj);
    }
  }
  return JSON.stringify(result);
}

// 3. ฟังก์ชันบันทึกและแก้ไขข้อมูล
function saveDeliveryData(itemStr) {
  var sheet = getDeliveriesSheet();
  var item = JSON.parse(itemStr);
  var data = sheet.getDataRange().getValues();
  
  // แปลง Array ชื่อคนส่ง เป็นข้อความคั่นด้วยลูกน้ำ เพื่อเซฟลงชีต
  var sendersStr = Array.isArray(item.senders) ? item.senders.join(',') : item.senders;
  
  var rowData = [
    item.id, item.type, item.month, item.date, item.agency, 
    item.plate, sendersStr, item.load, item.return, 
    item.price, item.actual, item.total, item.timestamp
  ];

  var found = false;
  // วนลูปหาว่าเคยมีบิลนี้ไหม ถ้ามีให้อัปเดต
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] === item.id) {
      sheet.getRange(i + 1, 1, 1, rowData.length).setValues([rowData]);
      found = true;
      break;
    }
  }
  
  // ถ้าไม่มีบิลนี้ ให้เพิ่มเป็นแถวใหม่
  if (!found) {
    sheet.appendRow(rowData);
  }
  return "Success";
}

// 4. ฟังก์ชันลบข้อมูล
function deleteDeliveryData(id) {
  var sheet = getDeliveriesSheet();
  var data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] === id) {
      sheet.deleteRow(i + 1);
      break;
    }
  }
  return "Success";
}
