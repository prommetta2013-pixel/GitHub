            if (amount === 0 || isNaN(amount)) return "ศูนย์บาทถ้วน";
            let numberStr = Math.abs(amount).toFixed(2).toString();
            let [integerPart, fractionalPart] = numberStr.split('.');
            let text = '';
            let digits = ['ศูนย์', 'หนึ่ง', 'สอง', 'สาม', 'สี่', 'ห้า', 'หก', 'เจ็ด', 'แปด', 'เก้า'];

            function convertPart(str) {
                let res = '';
                for(let i=0; i<str.length; i++) {
                    let d = parseInt(str[i]);
                    let pos = str.length - i - 1;

                        if(pos === 1 && d === 1) res += 'สิบ';

        function processSyncQueue() {
            if (isSyncing || syncQueue.length === 0) { updateSyncStatusUI(); return; }
            isSyncing = true; updateSyncStatusUI();
            const task = syncQueue.shift();
            let isDone = false; 

            const onSuccess = () => { if(isDone) return; isDone = true; isSyncing = false; processSyncQueue(); };
            const onFailure = (err) => { if(isDone) return; isDone = true; console.error("Sync Error:", err); isSyncing = false; processSyncQueue(); };

            setTimeout(() => { if (!isDone) onFailure("Timeout"); }, 8000);

            try {
                if (typeof google !== 'undefined' && google.script && google.script.run) {
                    const runner = google.script.run.withSuccessHandler(onSuccess).withFailureHandler(onFailure);
                    if (task.action === 'save' && runner.saveDebtorData) { runner.saveDebtorData(task.p1); } 
                    else if (task.action === 'status' && runner.updateDebtorStatus) { runner.updateDebtorStatus(task.p1, task.p2); } 
                    else if (task.action === 'delete' && runner.deleteDebtorData) { runner.deleteDebtorData(task.p1); } 
                    else { onSuccess(); } 
                } else {
                    setTimeout(onSuccess, 400); // Simulate network delay locally
                }
            } catch (e) { onFailure(e); }
        }

        // ================= DASHBOARD & DEBTORS =================
        function updateDashboard() {
            let uCount = 0, uSum = 0, pSum = 0, bSum = 0; let names = new Set();
            
            // ข้อมูลสำหรับวันปัจจุบัน (เพื่อรีเซ็ตรายวัน)
            const today = new Date();
            const yyyy = today.getFullYear();
            const mm = String(today.getMonth() + 1).padStart(2, '0');
            const dd = String(today.getDate()).padStart(2, '0');
            const todayStr = `${yyyy}-${mm}-${dd}`; // รูปแบบ YYYY-MM-DD
            
            let todayCash = 0;
            let todayDebt = 0;
            let todayBottles = 0;

            // ตารางสถิติพนักงานรายวัน
            const empStats = {};
            employeeData.forEach(e => { empStats[e.name] = { id: e.id, cash: 0, debt: 0 }; });

            // สรุปข้อมูลจากฐานข้อมูลลูกหนี้และรายบุคคล
            dbData.forEach(i => {
                // ข้อมูล All-Time (สะสมทั้งหมด)
                if (i.status === 'unpaid') { names.add(i.name.toLowerCase()); uSum += Number(i.amount); bSum += Number(i.bottles); } 
                else if (i.status === 'paid') { pSum += Number(i.amount); }

                // ข้อมูลประจำวัน (วันนี้)
                if (i.date === todayStr) {
                    if (i.status === 'paid') todayCash += Number(i.amount);
                    if (i.status === 'unpaid') todayDebt += Number(i.amount);
                    todayBottles += Number(i.bottles);

                    // เก็บยอดให้พนักงาน
                    if (i.seller && empStats[i.seller]) {
                        if (i.status === 'paid') empStats[i.seller].cash += Number(i.amount);
                        if (i.status === 'unpaid') empStats[i.seller].debt += Number(i.amount);
                    }
                }
            });

            // สรุปข้อมูลจากรถจัดส่งรวม (ถ้ามีการจัดส่งวันนี้)
            truckDeliveryData.forEach(i => {
                if (i.date === todayStr) {
                    todayCash += Number(i.totalCash);
                    todayDebt += (Number(i.botCredit) + Number(i.packCredit));
                    todayBottles += Number(i.botSold);

                    // หารเฉลี่ยยอดให้พนักงานที่มีชื่อในการจัดส่งนั้น
                    const senders = i.senders.split('+').map(s => s.trim());
                    senders.forEach(s => {
                        if (empStats[s]) {
                            empStats[s].cash += Math.floor(Number(i.totalCash) / senders.length);
                            empStats[s].debt += Math.floor((Number(i.botCredit) + Number(i.packCredit)) / senders.length);
                        }
                    });
                }
            });

            // อัปเดตหน้าจอ ข้อมูลสะสม
            document.getElementById('kpi-count').innerText = names.size; 
            document.getElementById('kpi-unpaid').innerText = `฿${uSum.toLocaleString()}`;
            document.getElementById('kpi-paid').innerText = `฿${pSum.toLocaleString()}`; 
            document.getElementById('kpi-bottles').innerText = bSum;
            
            // อัปเดตหน้าจอ ข้อมูลวันนี้ (รีเซ็ตเที่ยงคืนโดยอัตโนมัติ เพราะเช็คกับวันที่ปัจจุบัน)
            if(document.getElementById('kpi-today-cash')) document.getElementById('kpi-today-cash').innerText = `฿${todayCash.toLocaleString()}`;
            if(document.getElementById('kpi-today-debt')) document.getElementById('kpi-today-debt').innerText = `฿${todayDebt.toLocaleString()}`;
            if(document.getElementById('kpi-today-bottles')) document.getElementById('kpi-today-bottles').innerText = `${todayBottles} ถัง`;

            // อัปเดตตารางพนักงานรายวัน
            const empTableBody = document.getElementById('emp-daily-table-body');
            if (empTableBody) {
                empTableBody.innerHTML = '';
                let hasEmpData = false;
                Object.keys(empStats).forEach(name => {
                    const s = empStats[name];
                    empTableBody.innerHTML += `
                        <tr class="border-b last:border-0 hover:bg-gray-50 transition-colors">
                            <td class="p-3 text-xs font-bold text-gray-500 bg-gray-50 border-r w-24 text-center">${s.id}</td>
                            <td class="p-3 text-sm font-bold text-gray-800">${name}</td>
                            <td class="p-3 text-sm text-right font-bold text-green-700">฿${s.cash.toLocaleString()}</td>
                            <td class="p-3 text-sm text-right font-bold text-red-600">฿${s.debt.toLocaleString()}</td>
                        </tr>
                    `;
                    hasEmpData = true;
                });
                
                if (!hasEmpData || employeeData.length === 0) {
                    empTableBody.innerHTML = `<tr><td colspan="4" class="p-6 text-center text-gray-400 font-bold bg-gray-50"><i data-lucide="user-x" class="w-8 h-8 mx-auto mb-2 opacity-50"></i>ยังไม่มีข้อมูลพนักงาน</td></tr>`;
                    lucide.createIcons();
                }
            }

            renderCharts(uSum, pSum);
        }

        function exportCSV() {
            if(dbData.length === 0) { Swal.fire('Info', 'ไม่มีข้อมูลให้ส่งออก', 'info'); return; }
            let csv = '\uFEFF'; // BOM for Thai Excel

                csv += `${row.id},${safeName},${row.moo},${row.bottles},${row.amount},${row.date},${stat}\n`;
            });
            const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });

            document.body.appendChild(link); link.click(); document.body.removeChild(link);
        }
        function importCSV(e) {
            const file = e.target.files[0]; if(!file) return;
            const reader = new FileReader();
            reader.onload = function(evt) {
                const text = evt.target.result;
                const rows = text.split('\n');
                let imported = 0;
                for(let i=1; i<rows.length; i++) { // skip header
                    const line = rows[i].trim(); if(!line) continue;
                    // simple split handle (doesn't fully handle commas inside quotes for complex cases but good enough for this structure)
                    const cols = line.split(',');
                    if(cols.length >= 7) {
                        const id = cols[0].trim();
                        const name = cols[1].replace(/^"|"$/g, '').trim();
                        const moo = cols[2].trim();
                        const bottles = cols[3].trim();
                        const amount = cols[4].trim();
                        const date = cols[5].trim();
                        const stat = cols[6].trim() === 'ค้างชำระ' ? 'unpaid' : 'paid';
                        if(id && name) {
                            const idx = dbData.findIndex(d => d.id === id);
                            const item = { id, name, moo, bottles, amount, date, status: stat };
                            if(idx > -1) dbData[idx] = item; else dbData.push(item);
                            imported++;
                        }
                    }
                }
                addSyncTask('save', ''); // sync all (bulk not optimized but safe locally)
                updateDashboard(); renderDebtorsView(); updateCustomerDatalist();
                Swal.fire('สำเร็จ', `นำเข้าข้อมูลสำเร็จ ${imported} รายการ`, 'success');
                e.target.value = ''; // reset input
            };
            reader.readAsText(file);
        }

        function renderEmployeeDashboard() {
            const container = document.getElementById('employee-dashboard-cards'); if(!container) return;
            const currentMonth = new Date().toISOString().substring(0, 7); // 'YYYY-MM'
            const currentMonthData = truckDeliveryData.filter(d => (d.month || d.date.substring(0,7)) === currentMonth);
            
            // Map text senders to stats
            const stats = {};
            currentMonthData.forEach(d => {
                const names = d.senders.split('+').map(n => n.trim());
                names.forEach(n => {
                    if(!stats[n]) stats[n] = { botSold: 0, packSold: 0, totalCash: 0, trips: 0 };
                    stats[n].botSold += d.botSold; stats[n].packSold += d.packSold;
                    // Income is split equally among senders in the record for dashboard purposes
                    stats[n].totalCash += Math.floor(d.totalCash / names.length); 
                    stats[n].trips += 1;
                });
            });

            container.innerHTML = '';
            if(Object.keys(stats).length === 0) {
                container.innerHTML = `<div class="col-span-1 md:col-span-3 text-center text-sm text-gray-500 py-4">ยังไม่มีข้อมูลการส่งน้ำของเดือนนี้</div>`; return;
            }

            Object.keys(stats).sort((a,b) => stats[b].totalCash - stats[a].totalCash).forEach(name => {
                const s = stats[name];
                container.innerHTML += `
                <div class="bg-white rounded-xl p-4 shadow-sm border border-green-100 flex items-center gap-4 hover:shadow-md transition-shadow">
                    <div class="w-12 h-12 rounded-full bg-green-100 text-green-600 flex items-center justify-center shrink-0 font-bold text-xl">${name.charAt(0)}</div>
                    <div class="flex-1">
                        <h5 class="font-bold text-gray-800 text-base">${name}</h5>
                        <p class="text-[10px] text-gray-500 font-bold mb-1">ออกรถ ${s.trips} รอบ</p>
                        <div class="flex justify-between items-end">
                            <div class="text-xs text-gray-600 leading-tight">ถัง: <b class="text-blue-600">${s.botSold}</b><br>แพ็ค: <b class="text-blue-600">${s.packSold}</b></div>
                            <div class="text-right">
                                <span class="block text-[10px] text-gray-400">ยอดรับ(หารเฉลี่ย)</span>
                                <span class="font-bold text-green-600 text-sm">฿${s.totalCash.toLocaleString()}</span>
                            </div>
                        </div>
                    </div>
                </div>`;
            });
        }

        function updateGbDocNo() {
            const type = document.getElementById('gb-type').value; const docNoInput = document.getElementById('gb-doc-no');
            const date = document.getElementById('gb-date').value ? new Date(document.getElementById('gb-date').value) : new Date();
            const ym = `${date.getFullYear()}${String(date.getMonth() + 1).padStart(2, '0')}`;
            let seqData = JSON.parse(localStorage.getItem('freshy_doc_seq')) || { yearMonth: ym, DO: 0, RE: 0, DR: 0 };
            if (seqData.yearMonth !== ym) seqData = { yearMonth: ym, DO: 0, RE: 0, DR: 0 };
            let prefix = type === 'ใบส่งสินค้า' ? 'DO' : (type === 'ใบเสร็จรับเงิน' ? 'RE' : 'DR');
            docNoInput.value = `${prefix}${ym}${String(seqData[prefix] + 1).padStart(3, '0')}`;
        }
        document.getElementById('gb-date').addEventListener('change', updateGbDocNo);

        function consumeGbDocNo(type, dateStr) {
            const date = new Date(dateStr); const ym = `${date.getFullYear()}${String(date.getMonth() + 1).padStart(2, '0')}`;
            let seqData = JSON.parse(localStorage.getItem('freshy_doc_seq')) || { yearMonth: ym, DO: 0, RE: 0, DR: 0 };
            if (seqData.yearMonth !== ym) seqData = { yearMonth: ym, DO: 0, RE: 0, DR: 0 };
            let prefix = type === 'ใบส่งสินค้า' ? 'DO' : (type === 'ใบเสร็จรับเงิน' ? 'RE' : 'DR');
            seqData[prefix]++; localStorage.setItem('freshy_doc_seq', JSON.stringify(seqData));
            return `${prefix}${ym}${String(seqData[prefix]).padStart(3, '0')}`;
        }

        function gbProductSelected(selectEl, id) { const prod = productsData.find(p => p.code === selectEl.value); if(prod) { document.getElementById(`gb-name-${id}`).value = prod.name; document.getElementById(`gb-price-${id}`).value = prod.price; calcGbRow(id); } }
        function calcGbRow(id) { const qty = Number(document.getElementById(`gb-qty-${id}`).value) || 0; const price = Number(document.getElementById(`gb-price-${id}`).value) || 0; document.getElementById(`gb-total-${id}`).value = qty * price; calculateGbTotal(); }
        function removeGbItemRow(id) { const row = document.getElementById(`gb-row-${id}`); if(row) row.remove(); updateGbRowNumbers(); calculateGbTotal(); }
        function updateGbRowNumbers() { document.querySelectorAll('#gb-items-body tr').forEach((r, index) => { r.querySelector('.row-num').innerText = index + 1; }); }
        function calculateGbTotal() { let total = 0; document.querySelectorAll('[id^="gb-total-"]').forEach(input => { if(input.id !== 'gb-total-amount') total += Number(input.value) || 0; }); document.getElementById('gb-total-amount').value = total; }

        }

        function setupGbPrintModal(bill) {
            currentGbToPrint = bill; const info = loadBizInfo();
            document.getElementById('pgb-biz-name').innerText = info.name || 'ผู้ขาย'; document.getElementById('pgb-biz-loc').innerText = info.loc || '-'; document.getElementById('pgb-biz-tel').innerText = document.getElementById('conf-biz-tel') ? document.getElementById('conf-biz-tel').value : '-'; document.getElementById('pgb-biz-email').innerText = document.getElementById('conf-biz-email') ? document.getElementById('conf-biz-email').value : '-';
            document.getElementById('pgb-doc-type').innerText = bill.type; document.getElementById('pgb-doc-no').innerText = bill.docNo; const d = new Date(bill.date); document.getElementById('pgb-date').innerText = d.toLocaleDateString('th-TH', { day:'2-digit', month:'2-digit', year:'numeric' });
            document.getElementById('pgb-cus-name').innerText = bill.cusName; document.getElementById('pgb-cus-address').innerText = bill.cusAddress || '-'; document.getElementById('pgb-cus-tel').innerText = bill.cusTel || '-';

            const tbody = document.getElementById('pgb-items-body'); tbody.innerHTML = '';
            const MIN_ROWS = 10;
            bill.items.forEach((item, index) => {
                tbody.innerHTML += `
                    <tr style="height: 35px;">
                        <td style="border-right: 1px solid #000; padding: 6px 8px; text-align: center;">${index + 1}</td>
                        <td style="border-right: 1px solid #000; padding: 6px 8px; text-align: center;">${item.code !== '-' ? item.code : ''}</td>
                        <td style="border-right: 1px solid #000; padding: 6px 8px;">${item.name}</td>
                        <td style="border-right: 1px solid #000; padding: 6px 8px; text-align: center;">${item.qty}</td>
                        <td style="border-right: 1px solid #000; padding: 6px 8px; text-align: right;">${item.price.toLocaleString(undefined, {minimumFractionDigits: 2})}</td>
                        <td style="padding: 6px 8px; text-align: right;">${item.total.toLocaleString(undefined, {minimumFractionDigits: 2})}</td>
                    </tr>`;
            });
            const emptyRowsNeeded = MIN_ROWS - bill.items.length;
            if (emptyRowsNeeded > 0) {
                for(let i=0; i<emptyRowsNeeded; i++) {
                    tbody.innerHTML += `<tr style="height: 35px;"><td style="border-right: 1px solid #000; padding: 6px 8px;"></td><td style="border-right: 1px solid #000; padding: 6px 8px;"></td><td style="border-right: 1px solid #000; padding: 6px 8px;"></td><td style="border-right: 1px solid #000; padding: 6px 8px;"></td><td style="border-right: 1px solid #000; padding: 6px 8px;"></td><td style="padding: 6px 8px;"></td></tr>`;
                }
            }
            const formattedTotal = bill.total.toLocaleString(undefined, {minimumFractionDigits: 2});
            document.getElementById('pgb-total').innerText = formattedTotal; document.getElementById('pgb-net-total').innerText = formattedTotal;
            document.getElementById('pgb-text-total').innerText = ThaiBaht(bill.total);

            setCopyType('original'); openModal('general-bill-print-modal');
        }

        function loadBizInfo() {

            if(fdaInput) fdaInput.value = info.fda || '';
            const regInput = document.getElementById('conf-biz-reg');
            if(regInput) regInput.value = info.reg || '';

            const bizName = info.name || 'โรงน้ำดื่ม เฟรชชี่ วอเตอร์'; const elementsToUpdate = ['p-report-biz-name', 'form-biz-name', 'p-a6-biz-name', 'pt-header-biz-name', 'pts-header-biz-name', 'pgb-biz-name', 'pdf-biz-name', 'pdf-biz-name-2', 'pds-biz-name'];

            return info;
        }

        function syncData() {
            const isGasEnv = typeof google !== 'undefined' && google.script && google.script.run;
            updateSyncStatusUI(); 
            if (isGasEnv) {
                if (google.script.run.getDebtorsData) { google.script.run.withSuccessHandler(res => { if (res) { dbData = JSON.parse(res); localStorage.setItem('freshy_debt_data', res); } updateDashboard(); renderDebtorsView(); renderPaidView(); }).getDebtorsData(); }
                if (google.script.run.getDeliveriesData) { google.script.run.withSuccessHandler(res => { if (res) { const allDeliveries = JSON.parse(res); hospitalData = allDeliveries.filter(d => d.type === 'hospital'); saoData = allDeliveries.filter(d => d.type === 'sao'); localStorage.setItem('freshy_hospital_data', JSON.stringify(hospitalData)); localStorage.setItem('freshy_sao_data', JSON.stringify(saoData)); } renderHospitalView(); renderSaoView(); }).getDeliveriesData(); }
                if (google.script.run.getTruckData) { google.script.run.withSuccessHandler(res => { if(res) { truckDeliveryData = JSON.parse(res); localStorage.setItem('freshy_truck_data', res); } renderTruckSummaryList(); renderEmployeeDashboard(); }).getTruckData(); } 
                else { renderTruckSummaryList(); renderEmployeeDashboard(); }
                renderGbHistory(); 
            } else { updateDashboard(); renderDebtorsView(); renderPaidView(); renderHospitalView(); renderSaoView(); renderTruckSummaryList(); renderEmployeeDashboard(); renderGbHistory(); }
        }
